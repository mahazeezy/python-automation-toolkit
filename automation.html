from flask import Flask, request, jsonify import urllib.request from html.parser import HTMLParser app = Flask(__name__) @app.route("/") def home(): return open("automation.html").read() @app.route("/scrape") def scrape(): url = request.args.get("url", "") typ = request.args.get("type", "links") try: req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"}) with urllib.request.urlopen(req, timeout=8) as r: html = r.read().decode("utf-8", "ignore") class Parser(HTMLParser): def __init__(self): super().__init__() self.links, self.headings, self.images, self.texts = [], [], [], [] self.in_tag, self.cur_tag = False, "" def handle_starttag(self, tag, attrs): d = dict(attrs) if tag == "a" and "href" in d: self.links.append(d["href"]) if tag == "img" and "src" in d: self.images.append(d["src"]) if tag in ("h1","h2","h3"): self.in_tag = True; self.cur_tag = tag def handle_endtag(self, tag): if tag in ("h1","h2","h3"): self.in_tag = False def handle_data(self, data): if self.in_tag: self.headings.append(f"[{self.cur_tag.upper()}] {data.strip()}") self.texts.append(data) p = Parser(); p.feed(html) if typ == "links": items = p.links[:20]; label = "Links" elif typ == "headings": items = p.headings[:20]; label = "Headings" elif typ == "images": items = p.images[:20]; label = "Images" else: items = [" ".join(p.texts)[:500]]; label = "Text" if not items: return jsonify({"result": "No " + label + " found."}) return jsonify({"result": f"Found {len(items)} {label}:\n\n" + "\n".join(str(i) for i in items)}) except Exception as ex: return jsonify({"error": str(ex)}) @app.route("/check_urls", methods=["POST"]) def check_urls(): urls = request.json.get("urls", []) results = [] for url in urls[:10]: try: url = url.strip() if not url.startswith("http"): url = "https://" + url req = urllib.request.Request(url, headers={"User-Agent": "Mozilla/5.0"}) with urllib.request.urlopen(req, timeout=5) as r: results.append(f"✅ {url} — {r.status} OK") except Exception as ex: results.append(f"❌ {url} — {str(ex)[:50]}") return jsonify({"result": "\n".join(results)}) if __name__ == "__main__": app.run(debug=True)
